287inline bool operator==(const Variant &a, const Variant &b) {
288 if (a.get_kind() != b.get_kind()) return false;
289
290 switch (a.get_kind()) {
291 case Variant::INTEGER: return a.as_int() == b.as_int();
292 case Variant::NUMBER: return a.as_float() == b.as_float();
293 case Variant::BOOLEAN: return a.as_bool() == b.as_bool();
294 case Variant::STRING: return a.as_string() == b.as_string();
295 case Variant::NIL: return true;
296 case Variant::TABLE:
297 // Up to you: deep compare tables or just compare pointers.
298 return a.as_table() == b.as_table();
299 case Variant::BINARY: {
300 auto ba = a.as_binary(), bb = b.as_binary();
301 if (ba.size != bb.size) return false;
302 return std::memcmp(ba.start, bb.start, ba.size) == 0;
303 }
304 case Variant::UID: return a.as_uid().bytes == b.as_uid().bytes;
305 default: return false;
306 }
