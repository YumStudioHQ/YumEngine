263inline bool operator==(const Variant &a, const Variant &b) {
264 if (a.get_kind() != b.get_kind()) return false;
265
266 switch (a.get_kind()) {
267 case Variant::INTEGER: return a.as_int() == b.as_int();
268 case Variant::NUMBER: return a.as_float() == b.as_float();
269 case Variant::BOOLEAN: return a.as_bool() == b.as_bool();
270 case Variant::STRING: return a.as_string() == b.as_string();
271 case Variant::NIL: return true;
272 case Variant::TABLE:
273 // Up to you: deep compare tables or just compare pointers.
274 return a.as_table() == b.as_table();
275 case Variant::BINARY: {
276 auto ba = a.as_binary(), bb = b.as_binary();
277 if (ba.size != bb.size) return false;
278 return std::memcmp(ba.start, bb.start, ba.size) == 0;
279 }
280 default: return false;
281 }
