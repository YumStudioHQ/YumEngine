286inline bool operator==(const Variant &a, const Variant &b) {
287 if (a.get_kind() != b.get_kind()) return false;
288
289 switch (a.get_kind()) {
290 case Variant::INTEGER: return a.as_int() == b.as_int();
291 case Variant::NUMBER: return a.as_float() == b.as_float();
292 case Variant::BOOLEAN: return a.as_bool() == b.as_bool();
293 case Variant::STRING: return a.as_string() == b.as_string();
294 case Variant::NIL: return true;
295 case Variant::TABLE:
296 // Up to you: deep compare tables or just compare pointers.
297 return a.as_table() == b.as_table();
298 case Variant::BINARY: {
299 auto ba = a.as_binary(), bb = b.as_binary();
300 if (ba.size != bb.size) return false;
301 return std::memcmp(ba.start, bb.start, ba.size) == 0;
302 }
303 case Variant::UID: return a.as_uid().bytes == b.as_uid().bytes;
304 default: return false;
305 }
