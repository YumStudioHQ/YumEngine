38 inline size_t operator()(const YumEngine::Variant &v) const noexcept {
39 using Kind = YumEngine::Variant::VariantKind;
40 size_t h = static_cast<size_t>(v.get_kind());
41 auto mix = [](size_t x) {
42 uint64_t y = static_cast<uint64_t>(x);
43 y ^= (y >> 23);
44 y *= 0x2127599bf4325c37ULL;
45 y ^= (y >> 47);
46 return static_cast<size_t>(y);
47 };
48
49 switch (v.get_kind()) {
50 case Kind::INTEGER: return mix(h ^ std::hash<long long>()(v.as_int()));
51 case Kind::NUMBER: return mix(h ^ std::hash<double>()(v.as_float()));
52 case Kind::BOOLEAN: return mix(h ^ std::hash<bool>()(v.as_bool()));
53 case Kind::STRING: return mix(h ^ std::hash<std::string>()(v.as_string()));
54 case Kind::NIL: return mix(h);
55 case Kind::TABLE: return mix(h ^ std::hash<void*>()(v.as_table().get()));
56 case Kind::BINARY: {
57 auto bin = v.as_binary();
58 return mix(h ^ std::hash<size_t>()(bin.size) ^
59 (reinterpret_cast<size_t>(bin.start) >> 4));
60 }
61 case Kind::UID: return mix(h) ^ std::hash<uint64_t>()(v.as_uid().bytes);
62 default: return h;
63 }
64 }
65 };
