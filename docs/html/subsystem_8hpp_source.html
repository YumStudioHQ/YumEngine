157 inline static void ensureNamespace(lua_State *L, const std::string &ns) {
158 lua_getglobal(L, ns.c_str()); // push ns onto stack
159 if (lua_isnil(L, -1)) {
160 lua_pop(L, 1); // remove nil
161 lua_newtable(L); // create new table

163 lua_getglobal(L, ns.c_str()); // push it again
164 }
