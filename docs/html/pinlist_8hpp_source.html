67 it->object->free();

70 it->object_freed = true;

72 count++;
73 } else (*G_out()) << "yum: one element skipped" << std::endl;
74
75 it = it->org;
76 }
77

83 } else {
84 (*G_out()) << "yum: nothing to clean" << std::endl;
85 }

87 finalized = true;
