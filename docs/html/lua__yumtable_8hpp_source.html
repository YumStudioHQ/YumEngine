43 inline void push(lua_State *L, const Variant &key) {
44 switch (key.get_kind()) {
45 case Variant::INTEGER: lua_pushinteger(L, key.as_int()); break;
46 case Variant::NUMBER: lua_pushnumber(L, key.as_float()); break;
47 case Variant::BOOLEAN: lua_pushboolean(L, key.as_bool()); break;
48 case Variant::STRING: lua_pushstring(L, key.as_string().c_str()); break;
49 case Variant::BINARY: {
50 auto bin = key.as_binary();
51 lua_pushlstring(L, (const char*)bin.start, bin.size);
52 } break;
53 case Variant::TABLE: {
54 set_table(L, (*key.as_table())); // Variant::as_table() returns a std::shared_ptr<T>
55 } break;
56 case Variant::UID: push_uid(L, key.as_uid());
57 case Variant::NIL: lua_pushnil(L); break; // Todo?
58 default: break;
59 }
60 }

77 inline YumTable get_table(lua_State *L, int index) {
78 YumTable table;
79
80 // Absolute index in case the stack shifts
81 if (index < 0) index = lua_gettop(L) + index + 1;
82
83 lua_pushnil(L); // first key
84 while (lua_next(L, index) != 0) {
85
86 // value at -1, key at -2
87 Variant key = from_lua(L, -2);
88 Variant val = from_lua(L, -1);
89
90 table.set(key, val);
91
92 lua_pop(L, 1); // remove value, keep key for lua_next
93 }
94
95 return table;
96 }

98 inline Variant from_lua(lua_State *L, int idx) {
99 int type = lua_type(L, idx);
100
101 switch (type) {
102 case LUA_TNIL:
103 return Variant(); // NIL
104 case LUA_TBOOLEAN:
105 return Variant(static_cast<bool>(lua_toboolean(L, idx)));
106 case LUA_TNUMBER:
107 if (lua_isinteger(L, idx))
108 return Variant(static_cast<int64_t>(lua_tointeger(L, idx)));
109 else
110 return Variant(static_cast<double>(lua_tonumber(L, idx)));
111 case LUA_TSTRING: {
112 size_t len;
113 const char *str = lua_tolstring(L, idx, &len);
114 return Variant(std::string(str, len));
115 }
116 case LUA_TTABLE: {
117 // Recursively convert sub-table
118 YumTable sub = get_table(L, idx);
119
120 if (sub.has_key("__yumuid")) {
121 return Variant(YumUID{.bytes = sub.at("__yumuid").as_int()});
122 } else if (sub.has_key("__yum_type") && sub.at("__yum_type").as_string() == "binary") {
123 size_t len;
124 const char *str = lua_tolstring(L, idx, &len);
125 return Variant(YumBinaryBlob{.start = (const uint8_t*)str, .size = len});
126 }
127 return Variant(std::make_shared<YumTable>(std::move(sub)));
128 }
129 default:
130 // For userdata, lightuserdata, threads, functions, etc.
131 return Variant(); // NIL: NOT SUPPORTED BRO.
132 }
133 }
